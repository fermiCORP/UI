/**
 * Classe per gestire il login e la registrazione degli utenti.
 */
class LoginManager {
    /**
     * Crea un'istanza della classe LoginManager.
     * @param {HTMLElement} loginBtn - Il pulsante per il login.
     * @param {HTMLElement} registerBtn - Il pulsante per la registrazione.
     * @param {HTMLElement} loginForm - Il modulo di login.
     * @param {HTMLElement} registerForm - Il modulo di registrazione.
     */
    constructor(loginBtn, registerBtn, loginForm, registerForm) {
        this.loginBtn = loginBtn;
        this.registerBtn = registerBtn;
        this.loginForm = loginForm;
        this.registerForm = registerForm;

        // Aggiungi gli ascoltatori di eventi ai pulsanti di login e registrazione
        this.loginBtn.addEventListener('click', this.handleLoginButtonClick.bind(this));
        this.registerBtn.addEventListener('click', this.handleRegisterButtonClick.bind(this));
    }

    /**
     * Funzione per gestire il click sul pulsante di login.
     */
    handleLoginButtonClick() {
        this.loginBtn.style.backgroundColor = "#21264D";
        this.registerBtn.style.backgroundColor = "rgba(255 , 255 , 255, 0.2)";

        this.loginForm.style.left = "50%";
        this.registerForm.style.left = "-50%"

        this.loginForm.style.opacity = 1;
        this.registerForm.style.opacity = 0;

        document.querySelector(".col-1").style.borderRadius = "0 30% 20% 0";
    }

    /**
     * Funzione per gestire il click sul pulsante di registrazione.
     */
    handleRegisterButtonClick() {
        this.loginBtn.style.backgroundColor = "rgba(255 , 255, 255, 0.2)";
        this.registerBtn.style.backgroundColor = "#21264D";

        this.loginForm.style.left = "150%";
        this.registerForm.style.left = "50%"

        this.loginForm.style.opacity = 0;
        this.registerForm.style.opacity = 1;

        document.querySelector(".col-1").style.borderRadius = "0 20% 30% 0";
    }
}

// Elementi HTML
const loginBtn = document.querySelector("#login");
const registerBtn = document.querySelector("#register");
const loginForm = document.querySelector(".login-form");
const registerForm = document.querySelector(".register-form");

// Istanzia la classe LoginManager
const loginManager = new LoginManager(loginBtn, registerBtn, loginForm, registerForm);
